Here's a snapshot of the CC3D schema (may not be current): 

<?xml version="1.0"?>
<xs:schema xmlns:xs="http://www.w3.org/2001/XMLSchema"
	xmlns:jxb="http://java.sun.com/xml/ns/jaxb" jxb:version="2.0">
	<xs:annotation>
		<xs:appinfo>
      		<jxb:globalBindings
				fixedAttributeAsConstantProperty="true" collectionType="java.util.Vector"
				choiceContentProperty="false"
				enableFailFastCheck="false"
				generateIsSetMethod="false" underscoreBinding="asCharInWord" />
        </xs:appinfo>
	</xs:annotation>
	<xs:element name="CompuCell3D">
		<xs:complexType>
			<xs:sequence>
				<xs:element name="Potts" type="Potts" />
				<xs:element name="Plugin" type="Plugin" minOccurs="0"
					maxOccurs="unbounded" />
				<xs:element name="Steppable" type="Steppable" minOccurs="0"
					maxOccurs="unbounded" />
			</xs:sequence>
		</xs:complexType>
	</xs:element>
	
	
	<xs:complexType name="Potts">
		<xs:sequence>
            <xs:element name="Dimensions" type="Dimensions" />
            <xs:element name="Anneal" type="xs:int" />
            <xs:element name="Steps" type="xs:int" />
            <xs:element name="Temperature" type="xs:int" />
            <xs:element name="Flip2DimRatio" type="xs:int" />
            <xs:element name="Boundary_y" type="xs:string" />
            <xs:element name="NeighborOrder" type="xs:int"/>
            <xs:element name="RandomSeed" type="xs:int"/>
		</xs:sequence>
	</xs:complexType>
	
	<xs:complexType name="Dimensions">
		<xs:attribute name="x" type="xs:int" use="required"/>
		<xs:attribute name="y" type="xs:int" use="required"/>
		<xs:attribute name="z" type="xs:int" use="required"/>
	</xs:complexType>
	
		
	<xs:complexType name="Plugin">
	</xs:complexType>
	
	<xs:complexType name="Steppable">
	</xs:complexType>
	
		
	<xs:complexType name="Volume">
		<xs:complexContent>
			<xs:extension base="Plugin">
				<xs:sequence>
					<xs:element name="TargetVolume" type="xs:int" />
					<xs:element name="LambdaVolume" type="xs:double" />
				</xs:sequence>
				<xs:attribute name="Name" type="xs:NMTOKEN" use="required" fixed="Volume"/>
			</xs:extension>
		</xs:complexContent>
	</xs:complexType>

 
 	<xs:complexType name="CellType">
 		<xs:complexContent>
 			<xs:extension base="Plugin">
 				<xs:sequence>
 					<xs:element name="CellType" type="CellTypeItem" minOccurs="0" maxOccurs="unbounded"/>
 				</xs:sequence>
 				<xs:attribute name="Name" type="xs:NMTOKEN" use="required" fixed="CellType"/>
 			</xs:extension>
 		</xs:complexContent>
 	</xs:complexType>
	
	<xs:complexType name="CellTypeItem">
		<xs:attribute name="TypeName" type="CellTypeName" use="required" />
		<xs:attribute name="TypeId" type="xs:int" use="required" />
		<xs:attribute name="Freeze" type="xs:string" use="optional" />
	</xs:complexType>
	
	<xs:simpleType name="CellTypeName">
		<xs:restriction base="xs:string">
			<xs:enumeration value="Medium" />
			<xs:enumeration value="Fluid" />
			<xs:enumeration value="Wall" />
			<xs:enumeration value="Condensing" />
			<xs:enumeration value="NonCondensing" />
		</xs:restriction>
	</xs:simpleType>
	
	<xs:simpleType name="CCBoolean">
		<xs:restriction base="xs:string">
			<xs:enumeration value="yes" />
			<xs:enumeration value="no" />
		</xs:restriction>
	</xs:simpleType>
	
	<!--
    <Plugin Name="Contact"> 
      <Energy Type1="Foam" Type2="Foam">3</Energy>
      <Energy Type1="Medium" Type2="Medium">0</Energy> 
      <Energy Type1="Medium" Type2="Foam">0</Energy> 
      <Depth>2.1</Depth> 
	</Plugin>
	-->
	<xs:complexType name="Contact">
		<xs:complexContent>
			<xs:extension base="Plugin">
				<xs:sequence>
					<xs:element name="Energy" type="Energy" minOccurs="0" maxOccurs="unbounded" />
					<xs:element name="Depth" type="xs:int" />
					<xs:element name="NeighborOrder" type="xs:int"/>
				</xs:sequence>
				<xs:attribute name="Name" type="xs:NMTOKEN" use="required" fixed="Contact" />
			</xs:extension>
		</xs:complexContent>
	</xs:complexType>


  <xs:complexType name="Energy">
    <xs:simpleContent>
      <xs:extension base="xs:int">
      	<xs:attribute name="Type1" type="CellTypeName" use="required"/>
      	<xs:attribute name="Type2" type="CellTypeName" use="required"/>
      </xs:extension>     
    </xs:simpleContent>
  </xs:complexType>
  
    <!-- 
    <Plugin Name="Surface"> 
      <TargetSurface>20</TargetSurface> 
      <LambdaSurface>1.5</LambdaSurface> 
    </Plugin>  
    -->
	<xs:complexType name="Surface">
		<xs:complexContent>
			<xs:extension base="Plugin">
				<xs:sequence>
					<xs:element name="TargetSurface" type="xs:int"/>
					<xs:element name="LambdaSurface" type="xs:double"/>
				</xs:sequence>
				<xs:attribute name="Name" type="xs:NMTOKEN" use="required" fixed="Surface" />
			</xs:extension>
		</xs:complexContent>
	</xs:complexType>
	
	<!-- 
	<Plugin Name="VolumeFlex"> 
      <VolumeEnergyParameters CellType="Prestalk" TargetVolume="68" LambdaVolume="15"/> 
      <VolumeEnergyParameters CellType="Prespore" TargetVolume="69" LambdaVolume="12"/> 
      <VolumeEnergyParameters CellType="Autocycling" TargetVolume="80" LambdaVolume="10"/> 
      <VolumeEnergyParameters CellType="Ground" TargetVolume="0" LambdaVolume="0"/> 
      <VolumeEnergyParameters CellType="Wall" TargetVolume="0" LambdaVolume="0"/> 
    </Plugin>  
    -->
	<xs:complexType name="VolumeFlex">
		<xs:complexContent>
			<xs:extension base="Plugin">
				<xs:sequence>
					<xs:element name="VolumeEnergyParameters" type="VolumeEnergyParameters"
								minOccurs="0" maxOccurs="unbounded"/>
				</xs:sequence>
				<xs:attribute name="Name" type="xs:NMTOKEN" use="required" fixed="VolumeFlex"/>
			</xs:extension>
		</xs:complexContent>
	</xs:complexType>
	

	<xs:complexType name="VolumeEnergyParameters">
		<xs:attribute name="CellType" type="VolumeFlexCellType" />
	</xs:complexType>
	
	<xs:simpleType name="VolumeFlexCellType">
		<xs:restriction base="xs:string">
			<xs:enumeration value="Prestalk" />
			<xs:enumeration value="Prespore" />
			<xs:enumeration value="Autocycling" />
			<xs:enumeration value="Ground" />
			<xs:enumeration value="Wall" />
		</xs:restriction>
	</xs:simpleType>
	
	
	<!-- 
	<Steppable Type="BlobInitializer">
      <Gap>0</Gap>
      <Width>5</Width>
      <CellSortInit>yes</CellSortInit>   
      <Radius>40</Radius>
    </Steppable>
    -->
  	<xs:complexType name="BlobInitializer">
		<xs:complexContent>
			<xs:extension base="Steppable">
				<xs:sequence>
					<xs:element name="Gap" type="xs:int"/>
					<xs:element name="Width" type="xs:int"/>
					<xs:element name="CellSortInit" type="CCBoolean"/>
					<xs:element name="Radius" type="xs:int"/>
				</xs:sequence>
				<xs:attribute name="Type" type="xs:NMTOKEN" use="required" fixed="BlobInitializer"/>
			</xs:extension>
		</xs:complexContent>
	</xs:complexType>


</xs:schema>
